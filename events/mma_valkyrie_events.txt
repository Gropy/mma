	theme = faith
	left_portrait = {
		character = root
		animation = personality_zealous	
	}

	override_background = { event_background = temple_scope }

	immediate = {
		hidden_effect = {
				cp:councillor_court_chaplain = {
				save_scope_as = priest
			}
		}
	}

	option = { #odin
		name = valkyrie.0003.a

		duel = {
			skill = learning
			value = average_skill_rating
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 30
				}
				desc = odin.success
				send_interface_toast = {
					title = odin.success
					add_piety = 75
				}
				change_variable = {
					name = hunt_outcome
					add = 2
				}
				hidden_effect = {
					trigger_event = valkyrie.0005
				}
			}
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 1
					min = -8
				}
				desc = odin.fail
				send_interface_toast = {
					title = hunt.4003.c.tt_failure
					add_piety = -100
					add_trait = possessed_1
					}
				change_variable = {
					name = hunt_outcome
					add = -2
				}
			}
		}
		ai_chance = {
			base = 50
			modifier = {
				add = 50
				learning >= high_skill_rating
			}
		}
	}
		option = { #go back
		name = valkyrie.0003.b

		scope:activity = {
			complete_activity = no
		}

		ai_chance = {
			base = 0
		}
	}
}


valkyrie.0004 = { ###thor
	type = character_event
	title = valkyrie.0004.t
	desc = valkyrie.0004.desc
	theme = faith
	left_portrait = {
		character = root
		animation = personality_zealous	
	}

	override_background = { event_background = temple_scope }

	option = { #odin
		name = valkyrie.0004.a

		duel = {
			skill = martial
			value = average_skill_rating
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 30
				}
				desc = odin.success
				send_interface_toast = {
					title = odin.success
					add_prestige = 100
				}
				hidden_effect = {
					if = {
						limit = { 
							any_spouse = {
								is_ai = yes
							}
						}
						trigger_event = valkyrie.0007
					}
					else_if = {
						limit = { 
							any_child = {
								is_ai = yes
							}
						}
						trigger_event = valkyrie.0009
					}
					else_if = {
						trigger_event = valkyrie.0010
					}
				}
			}
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 1
					min = -8
				}
				desc = odin.fail
				send_interface_toast = {
					title = odin.fail
					add_prestige = -150
					add_trait = wounded_2	
				}
			}
		}

		ai_chance = {
			base = 50
			modifier = {
				add = 50
				martial >= high_skill_rating
			}
		}
	}

		option = { #go back
		name = valkyrie.0004.b

		scope:activity = {
			complete_activity = no
		}

		ai_chance = {
			base = 0
		}
	}
}
